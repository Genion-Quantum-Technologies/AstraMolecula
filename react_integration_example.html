React组件代码示例：

// MoleculeViewer3D.tsx
import React, { useEffect, useRef, useState } from 'react';
import { Dialog, DialogTitle, DialogContent, DialogActions, Button } from '@mui/material';
import { getDockingSDFFile } from '../../api/moleculeApi';

interface MoleculeViewer3DProps {
  open: boolean;
  onClose: () => void;
  taskId: string;
  filename: string;
  title: string;
}

export const MoleculeViewer3D: React.FC<MoleculeViewer3DProps> = ({
  open, onClose, taskId, filename, title
}) => {
  const viewerRef = useRef<HTMLDivElement>(null);
  const [loading, setLoading] = useState(false);
  const [viewer, setViewer] = useState<any>(null);

  useEffect(() => {
    if (open && taskId && filename) {
      loadMolecule();
    }
  }, [open, taskId, filename]);

  const loadMolecule = async () => {
    if (!viewerRef.current) return;
    
    setLoading(true);
    try {
      const sdfContent = await getDockingSDFFile(taskId, filename);
      
      const newViewer = window.$3Dmol.createViewer(viewerRef.current);
      newViewer.addModel(sdfContent, 'sdf');
      newViewer.setStyle({}, {
        stick: { colorscheme: 'Jmol', radius: 0.2 },
        sphere: { colorscheme: 'Jmol', scale: 0.3 }
      });
      newViewer.zoomTo();
      newViewer.render();
      
      setViewer(newViewer);
    } catch (error) {
      console.error('Error loading molecule:', error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <Dialog open={open} onClose={onClose} maxWidth="md" fullWidth>
      <DialogTitle>3D分子结构查看 - {title}</DialogTitle>
      <DialogContent>
        <div ref={viewerRef} style={{ height: '500px' }} />
      </DialogContent>
      <DialogActions>
        <Button onClick={onClose}>关闭</Button>
      </DialogActions>
    </Dialog>
  );
};
